package com.Erag0.ReversoContextBot.Loger;

import com.Erag0.ReversoContextBot.Loger.Behavior.LogBehavior;
import com.Erag0.ReversoContextBot.Loger.Behavior.SimpleLogBehavior;

public class Logger {
    private String pathToFile;
    private LogBehavior logBehaviour;

    public Logger(String pathToFile) {
        this.pathToFile = pathToFile;
        this.logBehaviour = new SimpleLogBehavior();
    }

    public void Log(Object obj) {
        String string = obj.toString();
        logBehaviour.write(this.pathToFile, string);
    }

    public void setPathToFile(String pathToFile) {
        this.pathToFile = pathToFile;
    }

    public void setLogBehaviour(LogBehavior logBehaviour) {
        this.logBehaviour = logBehaviour;
    }

}
